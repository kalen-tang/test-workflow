## 接口路径到方法名转换

### 转换算法

```python
def path_to_method_name(api_path: str, http_method: str = "POST") -> str:
    """
    将接口路径转换为controller方法名

    示例:
    - /dmb/nok9iy/activity/zazone/getVerifyCode → activity_za_zone_get_verify_code
    - /activity/zaZone/entryPage → activity_za_zone_entry_page
    - /reward/points/redeem → reward_points_redeem
    """
    # 1. 提取关键路径部分(跳过dmb/nok9iy等前缀)
    segments = [s for s in api_path.split('/') if s and s not in ['dmb', 'nok9iy', 'api', 'v1']]

    # 2. 处理驼峰命名: zaZone → za_zone
    def camel_to_snake(name):
        import re
        s1 = re.sub('(.)([A-Z][a-z]+)', r'\1_\2', name)
        return re.sub('([a-z0-9])([A-Z])', r'\1_\2', s1).lower()

    segments = [camel_to_snake(seg) for seg in segments]

    # 3. 拼接为snake_case
    method_name = '_'.join(segments)

    return method_name
```

| `/activity/zaZone/entryPage` | POST | `activity_za_zone_entry_page` |

| `/reward/points/redeem` | POST | `reward_points_redeem` |
| `/cuber/config/query` | GET | `cuber_config_query` |

## 方法名到Model类名转换

### 转换算法

```python
def method_to_model_name(method_name: str) -> str:
    """
    将snake_case方法名转换为PascalCase类名

    示例:
    - activity_za_zone_get_verify_code → ActivityZaZoneGetVerifyCode
    - activity_za_zone_entry_page → ActivityZaZoneEntryPage
    - reward_points_redeem → RewardPointsRedeem
    """
    # 1. 按下划线分割
    words = method_name.split('_')

    # 2. 每个单词首字母大写
    words = [word.capitalize() for word in words]

    # 3. 拼接
    class_name = ''.join(words)

    return class_name
```

```python
@allure.title("正常场景-已参与用户查询")
@allure.story("用户状态查询")
@pytest.mark.data('za_zone/entry_page_01_joined.yaml')
@pytest.mark.P0
def test_entry_page_01_joined(self, activity_sc, data):
    """
    测试场景: 已参与用户查询活动状态

    前置条件:
    - 用户已完成ZA Zone活动注册
    - 用户账号状态正常

    测试步骤:
    1. 构造已参与用户的请求参数(customer_no)
    2. 调用入口页查询接口
    3. 验证响应码为成功
    4. 验证返回活动状态信息

    预期结果:
    - code=IA0000
    - 返回用户参与信息
    - 包含活动状态和等级信息
    """
    # 步骤1: 获取测试数据
    step = data[0]

    # 步骤2: 构造请求
    user_id = step.get("user_id", "")
    header = {"userid": user_id, "language": "ca"}

    params = model.ActivityZaZoneEntryPage()
    params.json_data = {
        "customerNo": step.get("customer_no", "")
    }

    # 步骤3: 发送请求
    logging.info("测试已参与用户查询活动状态")
    resp = activity_sc.controller.activity_za_zone_entry_page(
        load=params, headers=header
    )

    # 步骤4: 验证响应码
    expected_code = step.get("expected_code", "IA0000")
    Assertion.str_assert(resp.json()["code"], expected_code)
```


## 响应断言模式

### 基础断言 (推荐)

只验证响应码:

```python
expected_code = step.get("expected_code", "000000")
Assertion.str_assert(resp.json()["code"], expected_code)
```

### 扩展断言 (可选)

验证特定业务字段:

```python
# 验证响应码
expected_code = step.get("expected_code", "000000")
Assertion.str_assert(resp.json()["code"], expected_code)

# 验证业务字段 (可选)
if expected_code == "000000":
    # 验证tokenId存在
    assert resp.json().get("data", {}).get("tokenId"), "tokenId不能为空"

    # 验证活动状态
    expected_status = step.get("expected_status", "ACTIVE")
    actual_status = resp.json()["data"]["status"]
    Assertion.str_assert(actual_status, expected_status)
```
